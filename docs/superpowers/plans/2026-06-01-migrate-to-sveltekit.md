# Migrate FlowTyping to SvelteKit Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Migrate FlowTyping from React/Next.js to SvelteKit (SPA-mode) with vanilla XState, vanilla Convex, Svelte 5 runes, and scoped CSS. Remove Tailwind, CVA, Zustand, and React-specific dependencies.

**Architecture:** SvelteKit runs as pure SPA (`ssr=false`, `adapter-static`) with file-based routing. XState machines and Vitest tests move unchanged. React hooks replaced with Svelte runes/stores. Tailwind utility classes converted to Svelte scoped `<style>` blocks. One global XState actor drives the app. One preferences store replaces Zustand.

**Tech Stack:** SvelteKit, Svelte 5 Runes, TypeScript, XState (vanilla), Convex (vanilla client), Vitest, ESLint + eslint-plugin-svelte, Storybook (@storybook/sveltekit)

```typescript
import type { Dictionary, Locale } from '$interfaces/types';
import en from '../../dictionaries/en.json';
import ru from '../../dictionaries/ru.json';

const dictionaries: Record<Locale, Dictionary> = { en, ru };

function getBrowserLocale(): Locale {
  if (typeof navigator === 'undefined') return 'en';
  const lang = navigator.language.slice(0, 2);
  return lang === 'ru' ? 'ru' : 'en';
}

function createLocaleStore() {
  const stored = typeof localStorage !== 'undefined' 
    ? localStorage.getItem('flow-typing-locale') 
    : null;
  
  let value = $state<Locale>(
    (stored as Locale) || getBrowserLocale()
  );

  $effect(() => {
    if (typeof localStorage !== 'undefined') {
      localStorage.setItem('flow-typing-locale', value);
    }
  });

  return {
    get value() { return value; },
    set value(v: Locale) { value = v; },
  };
}

export const locale = createLocaleStore();
export const dictionary = $derived(dictionaries[locale.value]);
```

- [ ] **Step 1: Create preferences store**

Create: `src/lib/preferences.svelte.ts`
```typescript
import { browser } from '$app/environment';
import type { KeyboardLayout } from '$interfaces/types';

interface SharedPreferences {
  exerciseId: string | null;
}

interface Preferences {
  locale: 'en' | 'ru';
  keyboardLayout: KeyboardLayout;
  shared: SharedPreferences;
  isInitialized: boolean;
}

const DEFAULT_PREFERENCES: Preferences = {
  locale: 'en',
  keyboardLayout: 'qwerty',
  shared: { exerciseId: null },
  isInitialized: false,
};

function deepMerge<T extends Record<string, unknown>>(
  target: T,
  source: Partial<T>
): T {
  const result = { ...target };
  for (const key in source) {
    if (source[key] !== null && typeof source[key] === 'object' && !Array.isArray(source[key])) {
      result[key] = deepMerge(
        result[key] as Record<string, unknown>,
        source[key] as Record<string, unknown>
      ) as T[Extract<keyof T, string>];
    } else if (source[key] !== undefined) {
      result[key] = source[key] as T[Extract<keyof T, string>];
    }
  }
  return result;
}

function loadPreferences(): Preferences {
  if (!browser) return DEFAULT_PREFERENCES;
  try {
    const stored = localStorage.getItem('flow-typing-user-preferences');
    if (stored) {
      const parsed = JSON.parse(stored) as Partial<Preferences>;
      return deepMerge(DEFAULT_PREFERENCES, parsed);
    }
  } catch {
    // ignore parse errors
  }
  return DEFAULT_PREFERENCES;
}

function createPreferencesStore() {
  const initial = loadPreferences();
  let prefs = $state<Preferences>({ ...initial, isInitialized: true });

  $effect(() => {
    if (browser) {
      const toStore = { ...prefs, isInitialized: undefined };
      localStorage.setItem('flow-typing-user-preferences', JSON.stringify(toStore));
    }
  });

  function update(partial: Partial<Preferences>) {
    prefs = deepMerge(prefs, partial);
  }

  function updateShared(partial: Partial<SharedPreferences>) {
    prefs = deepMerge(prefs, { shared: { ...prefs.shared, ...partial } });
  }

  return {
    get prefs() { return prefs; },
    update,
    updateShared,
  };
}

export const preferences = createPreferencesStore();
```
